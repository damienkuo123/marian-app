  <div class="card">

    <div id="setup-zone">

      <h2 style="color:var(--primary-blue);">歡迎來到口說挑戰！</h2>

      <p id="info-text">正在準備麥克風...</p>

      <button class="btn-start" id="start-btn" style="display:none;">我準備好了！</button>

    </div>


    <div id="studio" style="display:none;">

      <p style="color:var(--primary-blue); font-weight:bold;">🎙️ 第 <span id="idx">1</span>/10 題</p>

      <h1 id="word" style="font-size: 48px; color:#0d47a1; margin: 20px 0;">載入中...</h1>

      <button type="button" id="record-btn">🎤</button>

      <p id="status">按住錄音，放開評分</p>

      <div id="feedback"></div>

    </div>

  </div>


  <script>

    // --- ⚠️ 核心設定：請填入您的 GAS 網址 ---

    const GAS_URL = "https://script.google.com/macros/s/AKfycbxHkA5LmIqBVuMEkaT1IcOarqMuEtw7HBX-BplhvOiDw_XCGiksPQotRb0qwftm2Lub/exec"; 

    const AZURE_KEY = "3QzfGss1SpstM7DNIKKYtYDHEwCwB6rLhpKoqM1LGRI7QZjE4VptJQQJ99CCACYeBjFXJ3w3AAAaACOGvCso";

    const AZURE_REGION = "eastus";


    const params = new URLSearchParams(window.location.search);

    const STUDENT_ID = params.get('id') || "S001";


    let stream, mediaRecorder, chunks = [], words = [], idx = 0, total = 0;


    // 1. 初始化：請求麥克風並抓題目

    window.onload = async () => {

      try {

        stream = await navigator.mediaDevices.getUserMedia({ audio: true });

        document.getElementById('info-text').innerText = "麥克風已就緒！";

        document.getElementById('start-btn').style.display = 'inline-block';

      } catch (e) {

        alert("無法開啟麥克風，請檢查權限設定。");

      }

    };


    document.getElementById('start-btn').onclick = () => {

      document.getElementById('setup-zone').style.display = 'none';

      document.getElementById('studio').style.display = 'block';

      // 向 GAS 抓題目

      fetch(`${GAS_URL}?id=${STUDENT_ID}&action=get_data`)

        .then(res => res.json())

        .then(data => {

          words = data.words;

          renderWord();

        });

    };


    function renderWord() {

      if (idx < words.length) {

        document.getElementById('idx').innerText = idx + 1;

        document.getElementById('word').innerText = words[idx];

        document.getElementById('record-btn').style.display = 'block';

        document.getElementById('feedback').innerText = "";

        document.getElementById('status').innerText = "按住錄音，放開評分";

      } else {

        finishChallenge();

      }

    }


    // 2. 錄音邏輯 (按住錄音)

    const rb = document.getElementById('record-btn');

    rb.onpointerdown = (e) => {

      e.preventDefault(); chunks = [];

      mediaRecorder = new MediaRecorder(stream);

      mediaRecorder.ondataavailable = ev => chunks.push(ev.data);

      mediaRecorder.start();

      rb.classList.add('active');

      document.getElementById('status').innerText = "聆聽中...";

    };


    rb.onpointerup = () => {

      if (!mediaRecorder || mediaRecorder.state !== "recording") return;

      mediaRecorder.stop();

      rb.classList.remove('active');

      document.getElementById('status').innerText = "AI 評分中...";

      

      mediaRecorder.onstop = async () => {

        const audioBlob = new Blob(chunks, { type: 'audio/webm' });

        const reader = new FileReader();

        reader.readAsDataURL(audioBlob);

        reader.onloadend = async () => {

          const base64 = reader.result.split(',')[1];

          // 呼叫 Azure API (前端直接呼叫速度最快)

          const score = await evaluateAzure(base64, words[idx]);

          document.getElementById('feedback').innerText = "得分：" + Math.round(score);

          total += score; idx++;

          setTimeout(renderWord, 1200);

        };

      };

    };


    async function evaluateAzure(base64, text) {

      const url = `https://${AZURE_REGION}.stt.speech.microsoft.com/speech/recognition/conversation/cognitiveservices/v1?language=en-US`;

      const pronParams = { "ReferenceText": text, "GradingSystem": "HundredMark", "Granularity": "Phoneme", "Dimension": "Comprehensive" };

      const response = await fetch(url, {

        method: 'POST',

        headers: {

          'Ocp-Apim-Subscription-Key': AZURE_KEY,

          'Pronunciation-Assessment': btoa(JSON.stringify(pronParams)),

          'Content-Type': 'audio/webm; codecs=opus'

        },

        body: atob(base64).split('').map(c => c.charCodeAt(0))

      });

      const resData = await response.json();

      return resData.RecognitionStatus === "Success" ? resData.NBest[0].PronunciationAssessment.AccuracyScore : 0;

    }


    // 3. 結算並自動跳轉回點數頁面

    function finishChallenge() {

      const avg = Math.round(total / words.length);

      document.getElementById('studio').innerHTML = "<h3>挑戰完成！正在結算點數...</h3>";

      

      // 傳回 GAS 加分

      fetch(`${GAS_URL}?id=${STUDENT_ID}&action=add_score&score=${avg}`)

        .then(() => {

          alert("🎉 點數已自動存入！即將跳回首頁。");

          // 核心：自動飛回點數首頁

          window.location.href = `${GAS_URL}?id=${STUDENT_ID}`;

        });

    }

  </script>
