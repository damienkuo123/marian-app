let startTime; // 新增這行在變數區 rb.onpointerdown = (e) => { e.preventDefault(); startTime = Date.now(); // 記錄開始時間 chunks = []; mediaRecorder = new MediaRecorder(stream); mediaRecorder.ondataavailable = ev => chunks.push(ev.data); mediaRecorder.start(); rb.classList.add('active'); }; rb.onpointerup = () => { const duration = Date.now() - startTime; if (duration < 800) { // 如果錄音短於 0.8 秒 alert("唸太快了，請錄久一點點喔！"); mediaRecorder.stop(); rb.classList.remove('active'); return; } mediaRecorder.stop(); rb.classList.remove('active'); document.getElementById('status').innerText = "Gemini 老師正在細聽..."; mediaRecorder.onstop = () => { const audioBlob = new Blob(chunks, { type: 'audio/webm' }); const reader = new FileReader(); reader.readAsDataURL(audioBlob); reader.onloadend = () => { const base64 = reader.result.split(',')[1]; fetch(`${GAS_URL}?id=${STUDENT_ID}&action=evaluate_speech&word=${words[idx]}`, { method: 'POST', body: base64 }) .then(res => res.json()) .then(data => { // 如果拿到 1 分，代表 AI 聽到了但沒辦法給分 if (data.score === 1) { alert("AI 老師說：聽得不是很清楚，請對著麥克風大聲一點！"); document.getElementById('status').innerText = "請重錄一次"; } else { document.getElementById('feedback').innerText = "得分：" + data.score; total += data.score; idx++; setTimeout(renderWord, 1200); } }) .catch(err => alert("通訊失敗：" + err)); }; }; };
